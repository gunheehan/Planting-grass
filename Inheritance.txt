//상속 _Override 부모클래스
using System.Collections;
using System.Collections.Generic;
using UnityEngine;


abstract public class Gun_20211214 : MonoBehaviour { 
    protected string gunName;
    protected int gunage;

    protected virtual void info()
    {
        print("나는 인간입니다.");
    }

    abstract protected void Name();
}

//상속 + Override 
using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class test2 : Gun_20211214
{
    string schoolName;
    // Start is called before the first frame update
    void Start()
    {
        schoolName = "케이디 초등학교";
        gunName = "한건희";
        gunage = 26;

        info();
    }

    protected override void info()
    {
        base.info();
        print("나는 학생입니다.");
    }

    protected override void Name()
    {
        print(gunName);
    }
}
